### Linux（Ubuntu）

_相关教程：_ [MySQL - 下载与安装配置（Linux）](https://www.cnblogs.com/zhuchunlin/p/18562893)

    sudo apt-get update
    

    sudo apt-get install -y mysql-server mysql-client
    

    sudo vi /etc/mysql/mysql.conf.d/mysqld.cnf
    

找到\[mysqld\]选项组中的bind-address，然后使用i 键进入编辑模式，并输入#注释掉该行。修改的最终结果如下图最后一行所示。

![image](https://img2024.cnblogs.com/blog/3027557/202512/3027557-20251216184740806-1135866084.png)

    sudo systemctl status mysql #查看mysql服务的状态
    

    sudo systemctl start mysql #启动MySQL服务
    

    sudo systemctl stop mysql #停止MySQL服务
    

![image](https://img2024.cnblogs.com/blog/3027557/202512/3027557-20251216184846284-1413312583.png)

    sudo vim /etc/mysql/debian.cnf
    

    # MySQL客户端配置文件
    # 只包含客户端相关的配置选项
    
    [client]
    # 默认连接到本地MySQL
    host = localhost
    user = root
    socket = /var/run/mysqld/mysqld.sock
    
    [mysql]
    # 设置命令提示符
    prompt = (\u@\h) [\d]>\_
    # 启用自动补全
    auto-rehash = TRUE
    # 显示列名
    column-names = TRUE
    
    [mysqladmin]
    [mysqlcheck]
    [mysqldump]
    [mysqlimport]
    [mysqlshow]
    [mysqlslap]
    
    

![image](https://img2024.cnblogs.com/blog/3027557/202512/3027557-20251216185117506-12438712.png)

使用文件中的初始用户名和密码登录mysql

    mysql -u debian-sys-maint -p pfdKSjeFFhJtdGkr
    

    update mysql.user set authentication_string=password('123456') where user='root' and host='localhost';
    

**遇到的错误问题**

    # 第1分钟：检查服务状态
    sudo systemctl status mysql
    sudo systemctl restart mysql
    
    # 第2分钟：修复文件权限
    sudo chmod 644 /etc/mysql/debian.cnf
    
    # 第3分钟：登录系统账户
    mysql --defaults-extra-file=/etc/mysql/debian.cnf
    
    # 第4分钟：重置root密码（在MySQL中执行）
    ALTER USER 'root'@'localhost' IDENTIFIED BY '123456';
    FLUSH PRIVILEGES;
    exit;
    
    # 第5分钟：测试登录
    mysql -u root -p123456
