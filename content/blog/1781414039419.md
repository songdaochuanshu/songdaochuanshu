title: '让文字和标签的大小随着屏幕的尺寸做变化 等比缩放js'
date: "2021-09-16T06:49Z"
---
```js
    const fn = () => {
      /*rem*/
      let zj_width = 750; //设计稿尺寸
      /*让文字和标签的大小随着屏幕的尺寸做变化 等比缩放*/
      let html = document.querySelector("body");
      /*取到屏幕的宽度*/
      let width = window.innerWidth;
      /* 640 100  320 50 */
      let fontSize = (100 / zj_width / 10) * width;
      /*设置fontsize*/
      html.style.fontSize = fontSize + "px";
    };
    window.onresize = () => {
      fn();
    };
```
