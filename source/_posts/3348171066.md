    export default {
        data() {

            this.webViewUrl =
                \`http://192.168.101.25:5173/?data=${encodeURIComponent(this.url)}\` //pdf地址添加参数
        },
        onLoad() {

        },
        methods: {
            getMessage(data) {
       　　　　  uni.showToast({
                title: "reciveMessage接收到消息：" + JSON.stringify(data.detail),

            },
        },
    }
</script>
