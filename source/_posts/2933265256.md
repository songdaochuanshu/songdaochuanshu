*   **BufferedInputStreamзұ» дёҺ BufferedOutputStreamзұ»**
    ------------------------------------------------
    
    **BufferedInputStreamзұ»** еҸҜд»ҘеҜ№жүҖжңүInputStreamзҡ„еӯҗзұ»иҝӣиЎҢзј“еҶІеҢәзҡ„еҢ…иЈ…пјҢд»ҘиҫҫеҲ°**жҖ§иғҪзҡ„дјҳеҢ–**гҖӮ
    
    **BufferedOutputStreamзұ»** дёӯзҡ„ **flush()ж–№жі•** иў«з”ЁжқҘжҠҠзј“еҶІеҢәзҡ„еӯ—иҠӮ**еҶҷе…Ҙ**еҲ°ж–Үд»¶дёӯпјҢе№¶**жё…з©әзј“еӯҳ**гҖӮ
    
    ### **BufferedInputStreamзұ»зҡ„жһ„йҖ ж–№жі•пјҡ**
    
    жһ„йҖ ж–№жі•
    
    д»Ӣз»Қ
    
    **BufferedInputStream(FileInputStream fileInputStream);**
    
    еҲӣе»әдёҖдёӘ**еёҰжңү32дёӘеӯ—иҠӮзҡ„зј“еҶІиҫ“е…ҘжөҒ**гҖӮ
    
    **BufferedInputStream(FileInputStream fileInputStream , int size);**
    
    жҢү**жҢҮе®ҡзҡ„еӨ§е°ҸжқҘеҲӣе»әзј“еҶІиҫ“е…ҘжөҒ**гҖӮ
    
    ### **BufferedOutputStreamзұ»зҡ„жһ„йҖ ж–№жі•пјҡ**
    
    жһ„йҖ ж–№жі•
    
    д»Ӣз»Қ
    
    **BufferedOutputStream(FileOutputStream fileOutputStream);**
    
    еҲӣе»әдёҖдёӘ**еёҰжңү32дёӘеӯ—иҠӮзҡ„зј“еҶІиҫ“еҮәжөҒ**гҖӮ
    
    **BufferedOutputStream(FileOutputStream fileOutputStream , int size);**
    
    жҢү**жҢҮе®ҡзҡ„еӨ§е°ҸжқҘеҲӣе»әзј“еҶІиҫ“еҮәжөҒ**гҖӮ

        import java.io.*;
        
        public class Demo4 {
            public static void main(String[] args) {
                /**
                 * зј“еҶІеӯ—иҠӮиҫ“е…ҘжөҒпјҲBufferedInputStreamпјү

                File file = new File("C:\\JAVA_API_1.7дёӯж–Ү.chm");
                BufferedInputStream bufferedInputStream = null;//еҲӣе»әзј“еҶІеӯ—иҠӮжөҒ

                    fileInputStream=new FileInputStream(file);
                    bufferedInputStream = new BufferedInputStream(fileInputStream);//е°Ҷж–Үд»¶еӯ—иҠӮжөҒеҢ…иЈ…жҲҗзј“еҶІеӯ—иҠӮжөҒ

                    while ((bufferedInputStream.read(by))!=-1){//дҪҝз”Ёзј“еҶІеӯ—иҠӮжөҒиҜ»еҸ–ж•°жҚ®
        
                    }
                    long end = System.currentTimeMillis();//иҺ·еҫ—еҪ“еүҚжөҒз»“жқҹж—¶зҡ„жҜ«з§’еҖј
                    System.out.println("иҝҗиЎҢз»ҸеҺҶзҡ„жҜ«з§’ж•°пјҡ"+(end - stare));
        
                } catch (FileNotFoundException e) {
                    e.printStackTrace();
                } catch (IOException e) {
                    e.printStackTrace();
                } finally {
                    if (fileInputStream!=null){
                        try {
                            fileInputStream.close();
                        } catch (IOException e) {
                            e.printStackTrace();
                        }
                    }
                    if (bufferedInputStream!=null){
                        try {
                            bufferedInputStream.close();
                        } catch (IOException e) {
                            e.printStackTrace();
                        }
                    }

                 * зј“еҶІеӯ—иҠӮиҫ“еҮәжөҒпјҲBufferedOutputStreamпјү

                File file1 = new File("C:\\Word.txt");

                FileOutputStream fileOutputStream = null;
                try {
                    fileOutputStream=new FileOutputStream(file1);
                    bufferedOutputStream=new BufferedOutputStream(fileOutputStream);//е°Ҷж–Үд»¶иҫ“еҮәжөҒеҢ…иЈ…еҲ°зј“еҶІеӯ—иҠӮиҫ“еҮәжөҒ

                } catch (FileNotFoundException e) {
                    e.printStackTrace();
                } catch (IOException e) {
                    e.printStackTrace();
                } finally {
                    if (fileOutputStream!=null){
                        try {
                            fileOutputStream.close();
                        } catch (IOException e) {
                            e.printStackTrace();
                        }
                    }
                    if (bufferedOutputStream!=null){
                        try {
                            bufferedOutputStream.close();
                        } catch (IOException e) {
                            e.printStackTrace();
                        }
                    }
                }
            }

### **BufferedReaderзұ» дёҺ BufferedWriterзұ» е®һдҫӢпјҡ**

    import java.io.*;
    
    public class Demo6 {
        public static void main(String[] args) {
            File file = new File("C:\\Word.txt");
    
            /**
             * ж–Үд»¶зј“еҶІеӯ—з¬Ұиҫ“еҮәжөҒпјҲBufferedWriterпјү
             */
            FileWriter fileWriter = null;//еҲӣе»әж–Үд»¶еӯ—з¬Ұиҫ“еҮәжөҒ
            BufferedWriter bufferedWriter = null;//еҲӣе»әж–Үд»¶зј“еҶІеӯ—з¬Ұиҫ“еҮәжөҒ
    
            try {
                fileWriter = new FileWriter(file);
                bufferedWriter = new BufferedWriter(fileWriter);//е°Ҷж–Үд»¶еӯ—з¬Ұиҫ“еҮәжөҒеҢ…иЈ…жҲҗж–Үд»¶зј“еҶІеӯ—з¬Ұиҫ“еҮәжөҒ
    
                String str1 = "зҘһйҮҢ";
                String str2 = "з»«еҚҺ";
    
                bufferedWriter.write(str1);//з¬¬дёҖиЎҢж•°жҚ®
                bufferedWriter.newLine();//еҲӣе»әдёҖдёӘж–°иЎҢ
                bufferedWriter.write(str2);//з¬¬дәҢиЎҢж•°жҚ®
    
            } catch (IOException e) {
                e.printStackTrace();
            }finally {//<*> жіЁж„ҸпјҡжөҒзҡ„е…ій—ӯйЎәеәҸпјҢе…ҲеҲӣе»әзҡ„еҗҺе…ій—ӯгҖӮ <*>
                if (bufferedWriter!=null){
                    try {
                        bufferedWriter.close();
                    } catch (IOException e) {
                        e.printStackTrace();
                    }
                }
                if (fileWriter!=null){
                    try {
                        fileWriter.close();
                    } catch (IOException e) {
                        e.printStackTrace();
                    }
                }
            }
    
    
            /**
             * ж–Үд»¶зј“еҶІеӯ—з¬Ұиҫ“е…ҘжөҒпјҲBufferedReaderпјү
             */
            FileReader fileReader = null;
            BufferedReader bufferedReader = null;
            try {
                fileReader = new FileReader(file);
                bufferedReader = new BufferedReader(fileReader);//е°Ҷж–Үд»¶еӯ—з¬Ұиҫ“е…ҘжөҒеҢ…иЈ…жҲҗж–Үд»¶зј“еҶІеӯ—з¬Ұиҫ“е…ҘжөҒ
                String tmp = null;//дёҙж—¶еҸҳйҮҸ
                int i = 1;//и®Ўж•°еҷЁ
                while ((tmp = bufferedReader.readLine())!=null){//еҫӘзҺҜиҜ»еҸ–ж–Үд»¶дёӯзҡ„еҶ…е®№
                    System.out.println("з¬¬"+i+"иЎҢпјҡ"+tmp);
                    i++;
                }
            } catch (FileNotFoundException e) {
                e.printStackTrace();
            } catch (IOException e) {
                e.printStackTrace();
            } finally {
                if (bufferedReader!=null){
                    try {
                        bufferedReader.close();
                    } catch (IOException e) {
                        e.printStackTrace();
                    }
                }
                if (fileReader!=null){
                    try {
                        fileReader.close();
                    } catch (IOException e) {
                        e.printStackTrace();
                    }
                }
            }
        }
    }
