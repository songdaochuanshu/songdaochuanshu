        /// <summary>
        /// е®һзҺ°MinIOж–Үд»¶зҡ„дёҠдј 
        /// и°ғз”Ё AWSSDK.S3 е’Ң AWSSDK.Core 
        /// ж–Үд»¶зҡ„ ContentType и°ғз”Ё MimeMapping е®һзҺ°
        /// </summary>
        public class AmazonHelper {
            /// <summary>
            /// Bucketи®ҝй—®AccessKey
            /// </summary>
            readonly string AccessKey = "admin";
            /// <summary>
            /// Bucketи®ҝй—®SecretKey
            /// </summary>
            readonly string SecretKey = "123456+++";
            /// <summary>
            /// еӯҳеӮЁжЎ¶еҗҚз§°
            /// </summary>
            readonly string BucketName = "test";
            /// <summary>
            /// жңҚеҠЎең°еқҖ
            /// </summary>
            readonly string ServiceURL = "http://192.168.20.66:9000";
            /// <summary>
            /// йў„и§Ҳең°еқҖ
            /// </summary>
            readonly string WebURL = "https://res.test.cn";
            /// <summary>
            /// 
            /// </summary>
            //readonly string awsRegion = "us-east-1";
    
            readonly AmazonS3Client client;
            /// <summary>
            /// жһ„йҖ еҮҪж•°
            /// </summary>
            public AmazonHelper() {
                //жҸҗдҫӣawsAccessKeyIdе’ҢawsSecretAccessKeyжһ„йҖ еҮӯиҜҒ
                var credentials = new BasicAWSCredentials(AccessKey, SecretKey);
                //жҸҗдҫӣawsEndPointпјҲеҹҹеҗҚпјүиҝӣиЎҢи®ҝй—®й…ҚзҪ®
                var clientConfig = new AmazonS3Config {
                    // еҝ…йЎ»еңЁи®ҫзҪ®ServiceURLеүҚиҝӣиЎҢи®ҫзҪ®пјҢе№¶дё”йңҖиҰҒе’Ң`MINIO_REGION`зҺҜеўғеҸҳйҮҸдёҖиҮҙгҖӮ
                    //RegionEndpoint = RegionEndpoint.GetBySystemName(awsRegion),
                    RegionEndpoint = RegionEndpoint.USEast1,
                    // жӣҝжҚўжҲҗдҪ иҮӘе·ұзҡ„MinIO Serverзҡ„URL
                    ServiceURL = ServiceURL,
                    // еҝ…йЎ»и®ҫдёәtrue
                    ForcePathStyle = true,
                };
                client = new AmazonS3Client(credentials, clientConfig);
            }
    
            #region дёҠдј ж–Үд»¶
            /// <summary>
            /// дёҠдј ж–Үд»¶[жң¬ең°]
            /// </summary>
            /// <param name="key"></param>
            /// <param name="filePath">и·Ҝеҫ„</param>
            /// <returns>зҪ‘еқҖ</returns>
            public async Task<string> UploadFilePath(string saveKey, string filePath) {
                var request = new PutObjectRequest {
                    BucketName = BucketName,
                    Key = saveKey,
                    ContentType = GetContentType(saveKey),
                    FilePath = filePath,
                };
                var response = await client.PutObjectAsync(request);
                //
                return WebUrlKey(response.HttpStatusCode, saveKey);
            }
            /// <summary>
            /// дёҠдј ж–Үд»¶[жөҒ]
            /// </summary>
            public async Task<string> UploadInputStream(string saveKey, Stream inputStream) {
                var request = new PutObjectRequest {
                    BucketName = BucketName,
                    Key = saveKey,
                    ContentType = GetContentType(saveKey),
                    InputStream = inputStream //File.OpenRead
                };
                var response = await client.PutObjectAsync(request);
                //
                return WebUrlKey(response.HttpStatusCode, saveKey);
            }
            #endregion
    
            /// <summary>
            /// е®ўжҲ·з«ҜдёҠдј ж–Үд»¶
            /// </summary>
            public string GetPreSignedUrl(string saveKey) {
                var request = new GetPreSignedUrlRequest {
                    BucketName = BucketName,
                    Key = saveKey,
                    Expires = DateTime.UtcNow.AddMinutes(15),
                    Verb = HttpVerb.PUT,
                    ContentType = GetContentType(saveKey),
                    Protocol = Protocol.HTTP // жҢҮе®ҡдҪҝз”ЁHTTPSеҚҸи®®
                };
                var url= client.GetPreSignedURL(request);
                return url;
            }
    
            #region з§Ғжңүж–№жі•
            /// <summary>
            /// ж №жҚ®ж–Үд»¶Keyиҝ”еӣһйў„и§Ҳең°еқҖ
            /// </summary>
            private string WebUrlKey(HttpStatusCode statusCode, string saveKey) {
                if (statusCode != System.Net.HttpStatusCode.OK)
                    throw new Exception("дёҠдј ж–Үд»¶еӨұиҙҘ");
                return WebURL + "/" + BucketName + "/" + saveKey;
            }
    
            /// <summary>
            /// ж №жҚ®ж–Үд»¶иҺ·еҸ–ContentType
            /// </summary>
            private string GetContentType(string saveKey) {
                return MimeMapping.MimeUtility.GetMimeMapping(Path.GetFileName(saveKey));
            } 
            #endregion
        }
    }
